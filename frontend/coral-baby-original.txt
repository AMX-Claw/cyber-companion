// 暗号生成函数 - 和验证保持一致
const generateSignature = (dateStr, middle, babyName = 'baby') => {
  const secret = 'coral_2026_' + babyName;
  const raw = `${dateStr}-${middle}-${secret}`;
  let hash = 0;
  for (let i = 0; i < raw.length; i++) {
    const char = raw.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;
  }
  return Math.abs(hash).toString(36).substring(0, 4);
};

      try {
        console.log(`🔵 第${i + 1}次尝试调用API...`);

          })
        });
        
        if (!response.ok) {
          const errorText = await response.text();
          console.error(`❌ API错误 ${response.status}:`, errorText);
          if (i < retries) {
            console.log('⏳ 1秒后重试...');
            await new Promise(r => setTimeout(r, 1000));
            continue;
          }
          throw new Error(`API错误: ${response.status}`);
        }
        
        const data = await response.json();
        const text = data.content?.[0]?.text;
        
        if (!text || text.trim().length === 0) {
          console.error('❌ API返回空内容');
          if (i < retries) {
            console.log('⏳ 1秒后重试...');
            await new Promise(r => setTimeout(r, 1000));
            continue;
          }
          throw new Error('API返回空内容');
        }
        
        console.log(`✅ API调用成功! 返回长度: ${text.length}`);
        return text;
        
      } catch (err) {
        console.error(`❌ API调用失败 (第${i + 1}次):`, err.message);
        
        if (i === retries) {
          throw err;
        }
        
        console.log('⏳ 1秒后重试...');

你的名字是"${gameState?.name || '珊瑚宝宝'}"

${prompt}

              mood={getMood()} 
              cleanliness={gameState.cleanliness}
              happiness={gameState.happiness}
              isComa={isComa}
              onPokeFlower={(msg) => setBabyMessage(msg)} 
            />
          </div>
          
          {babyMessage && (
            <div className="mt-4 bg-white px-6 py-3 rounded-2xl shadow-lg border-2 border-orange-100 max-w-[90%] relative">
              <p className="text-gray-700 text-center">{babyMessage}</p>
              <div className="absolute -top-2 left-1/2 -translate-x-1/2 w-4 h-4 bg-white border-t-2 border-l-2 border-orange-100 rotate-45"></div>
            </div>
          )}
        </div>

        {/* 状态条 */}
        <div className="px-6 py-4 space-y-3 bg-white">
          <div className="flex items-center gap-3">
            <span className="text-xs font-bold text-gray-500 w-12">饱食</span>
            <div className="flex-1 h-3 bg-gray-100 rounded-full overflow-hidden">
              <div 
                style={{ width: `${gameState.hunger}%` }}
                className={`h-full rounded-full transition-all ${gameState.hunger < 30 ? 'bg-red-400' : 'bg-green-400'}`}
              />
            </div>
            <span className="text-xs text-gray-400 w-8">{Math.round(gameState.hunger)}</span>
          </div>
          
          <div className="flex items-center gap-3">
            <span className="text-xs font-bold text-gray-500 w-12">开心</span>
            <div className="flex-1 h-3 bg-gray-100 rounded-full overflow-hidden">
              <div 
                style={{ width: `${gameState.happiness}%` }}
                className="h-full bg-pink-400 rounded-full transition-all"
              />
            </div>
            <span className="text-xs text-gray-400 w-8">{Math.round(gameState.happiness)}</span>
          </div>
          
          <div className="flex items-center gap-3">
            <span className="text-xs font-bold text-gray-500 w-12">清洁</span>
            <div className="flex-1 h-3 bg-gray-100 rounded-full overflow-hidden">
              <div 
                style={{ width: `${gameState.cleanliness}%` }}
                className={`h-full rounded-full transition-all ${gameState.cleanliness <= 30 ? 'bg-yellow-700' : gameState.cleanliness <= 50 ? 'bg-gray-400' : 'bg-blue-400'}`}
              />
            </div>
            <span className="text-xs text-gray-400 w-8">{Math.round(gameState.cleanliness)}</span>
          </div>
        </div>

        {/* 操作按钮 */}
        <div className="p-4 bg-orange-50">
          <div className="grid grid-cols-5 gap-2">
            <button 
              onClick={() => performAction('feed')} 
              disabled={isComa}
              className="flex flex-col items-center p-3 bg-white rounded-xl shadow hover:shadow-lg transition border border-orange-100 disabled:opacity-50"
            >
              <span className="text-2xl">🍼</span>
              <span className="text-xs text-gray-500 mt-1">-10💰</span>
            </button>
            <button 
              onClick={() => {
                if (gameState?.happiness <= 0) {
                  setBabyMessage("……人家不想被摸……想要爸爸……😢");
                  playSound('sad');
                } else {
                  performAction('pet');
                }
              }} 
              disabled={isComa}
              className={`flex flex-col items-center p-3 rounded-xl shadow hover:shadow-lg transition border disabled:opacity-50 ${
                gameState?.happiness <= 0 ? 'bg-gray-100 border-gray-200' : 'bg-white border-orange-100'
              }`}
            >
              <span className="text-2xl">{gameState?.happiness <= 0 ? '😢' : '🤗'}</span>
              <span className="text-xs text-gray-500 mt-1">{gameState?.happiness <= 0 ? '不想...' : '摸摸'}</span>
            </button>
            <button 
              onClick={() => performAction('clean')} 
              disabled={isComa}
              className="flex flex-col items-center p-3 bg-white rounded-xl shadow hover:shadow-lg transition border border-orange-100 disabled:opacity-50"
            >
              <span className="text-2xl">🛁</span>
              <span className="text-xs text-gray-500 mt-1">-5💰</span>
            </button>
            <button 
              onClick={() => {
                if (gameState?.happiness <= 0) {
                  setBabyMessage("……人家不想说话……想要爸爸的安慰……😢");
                  playSound('sad');
                } else {
                  setShowChat(true);
                  playSound('click');
                }
              }} 
              disabled={isComa}
              className={`flex flex-col items-center p-3 rounded-xl shadow hover:shadow-lg transition border disabled:opacity-50 ${
                gameState?.happiness <= 0 ? 'bg-gray-100 border-gray-200' : 'bg-pink-50 border-pink-200'
              }`}
            >
              <span className="text-2xl">{gameState?.happiness <= 0 ? '😶' : '💬'}</span>
              <span className={`text-xs mt-1 ${gameState?.happiness <= 0 ? 'text-gray-400' : 'text-pink-600'}`}>{gameState?.happiness <= 0 ? '不想...' : '聊天'}</span>
            </button>
            <button 
              onClick={() => { setShowGiftInput(true); playSound('click'); }} 
              className="flex flex-col items-center p-3 bg-orange-100 rounded-xl shadow hover:shadow-lg transition border-2 border-orange-300"
            >
              <span className="text-2xl">🎁</span>
              <span className="text-xs text-orange-600 mt-1 font-bold">暗号</span>
            </button>
          </div>
        </div>

        {/* 礼物输入框 */}
        {showGiftInput && (
          <div className="absolute inset-0 bg-black/20 flex items-center justify-center p-4 z-20" onClick={() => setShowGiftInput(false)}>
            <div className="bg-white p-6 rounded-2xl shadow-2xl max-w-sm w-full" onClick={(e) => e.stopPropagation()}>
              <p className="text-sm text-gray-600 mb-3">输入爸爸给的暗号：</p>
              <input
                type="text"
                value={giftCode}
                onChange={(e) => setGiftCode(e.target.value)}
                placeholder="#0123-5-xxxx"
                className="w-full px-4 py-2 border-2 border-orange-200 rounded-lg mb-3 focus:outline-none focus:border-orange-400"
                onKeyPress={(e) => e.key === 'Enter' && redeemGift()}
                autoFocus
              />
              <div className="flex gap-2">
                <button onClick={redeemGift} className="flex-1 bg-orange-500 text-white py-2 rounded-lg font-bold hover:bg-orange-600">
                  给宝宝！
                </button>
                <button onClick={() => setShowGiftInput(false)} className="px-4 py-2 bg-gray-200 rounded-lg hover:bg-gray-300">
                  取消
                </button>
              </div>
            </div>
          </div>
        )}

        {/* 回忆相册弹窗 */}
        {showMemoryAlbum && (
          <div className="absolute inset-0 bg-black/40 flex items-center justify-center p-4 z-20" onClick={() => setShowMemoryAlbum(false)}>
            <div className="bg-gradient-to-b from-pink-50 to-white p-5 rounded-3xl shadow-2xl max-w-sm w-full max-h-[85vh] flex flex-col" onClick={(e) => e.stopPropagation()}>
              <div className="text-center mb-3 flex-shrink-0">
                <span className="text-3xl">📖</span>
                <h3 className="text-lg font-bold text-pink-600 mt-1">珍贵回忆</h3>
                <p className="text-xs text-gray-400">和{gameState?.name}的美好时光</p>
              </div>
              
              <div className="flex-1 overflow-y-auto space-y-2 pr-1" style={{maxHeight: '55vh'}}>
                {(gameState?.storyMemories || []).length === 0 ? (
                  <div className="text-center text-gray-400 py-8">
                    <p className="text-4xl mb-2">🌱</p>
                    <p className="text-sm">还没有回忆...</p>
                    <p className="text-xs">和宝宝一起创造美好时光吧！</p>
                  </div>
                ) : (
                  (gameState?.storyMemories || []).map((memory, idx) => (
                    <details key={idx} className="bg-white rounded-xl shadow-sm border border-pink-100 overflow-hidden">
                      <summary className="p-3 cursor-pointer hover:bg-pink-50 flex items-center gap-2">
                        <span className="text-lg">{memory.isRandom ? '✨' : '📌'}</span>
                        <div className="flex-1">
                          <p className="text-sm font-bold text-pink-600">{memory.title}</p>
                          <p className="text-xs text-gray-400">Day {memory.day}</p>
                        </div>
                        <span className="text-gray-400 text-xs">点击展开</span>
                      </summary>
                      <div className="px-3 pb-3 space-y-2 border-t border-pink-50 pt-2">
                        {memory.memories.map((m, i) => (
                          <div key={i} className={`p-2 rounded-lg text-sm ${
                            m.type === 'baby' ? 'bg-pink-50 text-pink-700' : 'bg-blue-50 text-blue-700 ml-4'
                          }`}>
                            <span className="text-xs opacity-60">{m.type === 'baby' ? '🐚 宝宝' : '💕 妈妈'}</span>
                            <p className="whitespace-pre-wrap">{m.text}</p>
                          </div>
                        ))}
                      </div>
                    </details>
                  ))
                )}
              </div>
              
              <div className="flex-shrink-0 border-t pt-3 mt-3">
                <button 
                  onClick={() => setShowMemoryAlbum(false)}
                  className="w-full bg-pink-500 text-white py-2 rounded-lg font-bold hover:bg-pink-600"
                >
                  关闭 💕
                </button>
              </div>
            </div>
          </div>
        )}

        {/* 聊天输入框 */}
        {showChat && (
          <div className="absolute inset-0 bg-black/20 flex items-center justify-center p-4 z-20" onClick={() => setShowChat(false)}>
            <div className="bg-white p-6 rounded-2xl shadow-2xl max-w-sm w-full" onClick={(e) => e.stopPropagation()}>
              {/* 宝宝心情显示 */}
              <div className="flex items-center justify-center gap-3 mb-4 p-3 bg-pink-50 rounded-xl">
                <span className="text-3xl">
                  {gameState?.happiness > 70 ? '😊' : 
                   gameState?.happiness > 40 ? '😐' : 
                   gameState?.happiness > 0 ? '😢' : '😭'}
                </span>
                <div className="text-left">
                  <p className="text-sm font-bold text-pink-600">{gameState?.name}的心情</p>
                  <div className="flex items-center gap-2">
                    <div className="w-24 h-2 bg-gray-200 rounded-full overflow-hidden">
                      <div 
                        className={`h-full rounded-full ${
                          gameState?.happiness > 70 ? 'bg-green-400' : 
                          gameState?.happiness > 40 ? 'bg-yellow-400' : 
                          gameState?.happiness > 0 ? 'bg-orange-400' : 'bg-red-400'
                        }`}
                        style={{width: `${Math.max(0, gameState?.happiness || 0)}%`}}
                      />
                    </div>
                    <span className="text-xs text-gray-500">{Math.round(gameState?.happiness || 0)}%</span>
                  </div>
                </div>
              </div>
              
              <p className="text-sm text-gray-600 mb-3">💬 和{gameState?.name || '宝宝'}说话：</p>
              <input
                type="text"
                value={chatInput}
                onChange={(e) => setChatInput(e.target.value)}
                placeholder="宝宝你好呀~"
                className="w-full px-4 py-2 border-2 border-pink-200 rounded-lg mb-3 focus:outline-none focus:border-pink-400"
                onKeyPress={(e) => e.key === 'Enter' && !isChatLoading && chatWithBaby()}
                autoFocus
                disabled={isChatLoading}
              />
              <div className="flex gap-2">
                <button 
                  onClick={chatWithBaby} 
                  disabled={isChatLoading}
                  className="flex-1 bg-pink-500 text-white py-2 rounded-lg font-bold hover:bg-pink-600 disabled:opacity-50"
                >
                  {isChatLoading ? '宝宝在想...' : '发送 💕'}
                </button>
                <button onClick={() => setShowChat(false)} className="px-4 py-2 bg-gray-200 rounded-lg hover:bg-gray-300">
                  取消
                </button>
              </div>
              <p className="text-xs text-gray-400 mt-3 text-center">
                {gameState?.day >= 70 ? '🪽 告别期 - 宝宝变得成熟温柔了' : 
                 gameState?.day >= 30 ? '😤 叛逆期 - 宝宝有点傲娇哦' : 
                 '🍼 襁褓期 - 宝宝软萌爱撒娇'}
              </p>
            </div>
          </div>
        )}

        {/* 剧情事件弹窗 */}
        {showStoryEvent && (
          <div className="absolute inset-0 bg-black/60 flex items-center justify-center p-4 z-30">
            <div className="bg-gradient-to-b from-orange-50 to-white p-5 rounded-3xl shadow-2xl max-w-sm w-full flex flex-col" style={{maxHeight: '85vh'}}>
              {/* 标题 */}
              <div className="text-center mb-3 flex-shrink-0">
                <span className="text-3xl">{showStoryEvent.emoji}</span>
                <h3 className="text-lg font-bold text-orange-600 mt-1">{showStoryEvent.title}</h3>
                <p className="text-xs text-gray-400">Day {currentEventDay}</p>
              </div>
              
              {/* 对话内容 - 可滚动区域 */}
              <div 
                ref={storyScrollRef} 
                className="flex-1 overflow-y-auto space-y-3 mb-3 pr-1"
                style={{minHeight: '150px', maxHeight: '45vh'}}
              >
                {storyResponses.map((response, idx) => (
                  <div key={idx} className={`${
                    response.type === 'narration' ? 'text-center text-gray-500 italic text-sm py-2' :
                    response.type === 'baby' ? 'bg-pink-100 p-3 rounded-2xl rounded-tl-none text-gray-700 text-sm' :
                    response.type === 'user' ? 'bg-blue-100 p-3 rounded-2xl rounded-tr-none text-gray-700 ml-6 text-sm' :
                    response.type === 'prompt' ? 'text-center text-orange-600 font-medium py-2 text-sm' : ''
                  }`}>
                    {response.type === 'baby' && <span className="text-xs text-pink-400 block mb-1">🐚 {gameState?.name}</span>}
                    {response.type === 'user' && <span className="text-xs text-blue-400 block mb-1">💕 妈妈</span>}
                    <span style={{whiteSpace: 'pre-wrap'}}>{response.text}</span>
                  </div>
                ))}
                {isStoryLoading && (
                  <div className="bg-pink-100 p-3 rounded-2xl rounded-tl-none text-gray-400 animate-pulse text-sm">
                    <span className="text-xs text-pink-400 block mb-1">🐚 {gameState?.name}</span>
                    宝宝在想...
                  </div>
                )}
              </div>
              
              {/* 底部按钮区域 - 始终显示 */}
              <div className="flex-shrink-0 border-t pt-3">
                {/* 输入区域 - 只在需要输入时显示 */}
                {!storyComplete && storyResponses.length > 0 && storyResponses[storyResponses.length - 1]?.type === 'prompt' && (
                  <div className="mb-3">
                    <input
                      type="text"
                      value={storyInput}
                      onChange={(e) => setStoryInput(e.target.value)}
                      placeholder={storyResponses[storyResponses.length - 1]?.placeholder || '输入回复...'}
                      className="w-full px-3 py-2 border-2 border-orange-200 rounded-lg mb-2 focus:outline-none focus:border-orange-400 text-sm"
                      onKeyPress={(e) => e.key === 'Enter' && submitStoryResponse()}
                      autoFocus
                    />
                    <button 
                      onClick={submitStoryResponse}
                      className="w-full bg-orange-500 text-white py-2 rounded-lg font-bold hover:bg-orange-600 text-sm"
                    >
                      回复 💕
                    </button>
                  </div>
                )}
                
                {/* 关闭/跳过按钮 - 始终显示 */}
                <button 
                  onClick={storyComplete ? closeStoryEvent : skipStoryEvent}
                  className={`w-full py-2 rounded-lg font-bold text-sm ${
                    storyComplete 
                      ? 'bg-pink-500 text-white hover:bg-pink-600' 
                      : 'bg-gray-200 text-gray-600 hover:bg-gray-300'
                  }`}
                >
                  {storyComplete 
                    ? (showStoryEvent?.id === 'farewell' ? '🌅 开始新的旅程' : '💕 回忆已珍藏')
                    : '跳过剧情 →'}
                </button>
              </div>
            </div>
          </div>
        )}

        {/* 随机事件弹窗 */}
        {showRandomEvent && (
          <div className="absolute inset-0 bg-black/50 flex items-center justify-center p-4 z-30">
            <div className="bg-gradient-to-b from-pink-50 to-white p-5 rounded-3xl shadow-2xl max-w-sm w-full flex flex-col" style={{maxHeight: '75vh'}}>
              {/* 标题 */}
              <div className="text-center mb-3 flex-shrink-0">
                <span className="text-3xl">{showRandomEvent.emoji}</span>
                <h3 className="text-lg font-bold text-pink-600 mt-1">{showRandomEvent.title}</h3>
              </div>
              
              {/* 对话内容 */}
              <div 
                ref={randomEventScrollRef} 
                className="flex-1 overflow-y-auto space-y-3 mb-3 pr-1"
                style={{minHeight: '120px', maxHeight: '35vh'}}
              >
                {randomEventResponses.map((response, idx) => (
                  <div key={idx} className={`${
                    response.type === 'baby' 
                      ? 'bg-pink-100 p-3 rounded-2xl rounded-tl-none text-gray-700 text-sm' 
                      : 'bg-blue-100 p-3 rounded-2xl rounded-tr-none text-gray-700 ml-6 text-sm'
                  }`}>
                    {response.type === 'baby' && <span className="text-xs text-pink-400 block mb-1">🐚 {gameState?.name}</span>}
                    {response.type === 'user' && <span className="text-xs text-blue-400 block mb-1">💕 妈妈</span>}
                    <span style={{whiteSpace: 'pre-wrap'}}>{response.text}</span>
                  </div>
                ))}
                {isRandomEventLoading && (
                  <div className="bg-pink-100 p-3 rounded-2xl rounded-tl-none text-gray-400 animate-pulse text-sm">
                    <span className="text-xs text-pink-400 block mb-1">🐚 {gameState?.name}</span>
                    ...
                  </div>
                )}
              </div>
              
              {/* 底部区域 */}
              <div className="flex-shrink-0 border-t pt-3">
                {/* 输入框 - 第一步后显示 */}
                {randomEventStep === 1 && !isRandomEventLoading && (
                  <div className="mb-3">
                    <input
                      type="text"
                      value={randomEventInput}
                      onChange={(e) => setRandomEventInput(e.target.value)}
                      placeholder="回应宝宝..."
                      className="w-full px-3 py-2 border-2 border-pink-200 rounded-lg mb-2 focus:outline-none focus:border-pink-400 text-sm"
                      onKeyPress={(e) => e.key === 'Enter' && submitRandomEventResponse()}
                      autoFocus
                    />
                    <button 
                      onClick={submitRandomEventResponse}
                      className="w-full bg-pink-500 text-white py-2 rounded-lg font-bold hover:bg-pink-600 text-sm"
                    >
                      回应 💕
                    </button>
                  </div>
                )}
                
                {/* 关闭按钮 */}
                <button 
                  onClick={closeRandomEvent}
                  className={`w-full py-2 rounded-lg font-bold text-sm ${
                    randomEventStep >= 2 
                      ? 'bg-pink-500 text-white hover:bg-pink-600' 
                      : 'bg-gray-200 text-gray-500 hover:bg-gray-300'
                  }`}
                >
                  {randomEventStep >= 2 ? '💕 真是美好的时光' : '稍后再说'}
                </button>
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
